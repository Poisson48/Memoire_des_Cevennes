# Mémoire des Cévennes
# robots.txt : règle ce que les moteurs de recherche peuvent crawler.
# Les pages privées (login, mon-compte, admin…) portent un meta noindex
# dans leur HTML : pas besoin de les bloquer ici (laisser Google les
# crawler permet de respecter ce noindex).

User-agent: *
Allow: /

# API et données brutes : pas indexable utilement.
Disallow: /api/
Disallow: /data/

# Médias uploadés : la visibilité (membres / public) n'est pas filtrée
# au niveau du fichier statique, on bloque par prudence pour ne pas
# exposer des récits réservés aux membres dans Google Images.
Disallow: /uploads/

# Console d'admin : aucune raison d'aller fouiller.
Disallow: /admin
Disallow: /admin.html

# ----------------------------------------------------------------------
# Robots d'IA et aspirateurs de contenu : interdiction totale.
# Le site reste ouvert aux vrais moteurs de recherche (Googlebot, Bingbot,
# DuckDuckBot...) pour rester indexable, mais refuse les robots qui aspirent
# le contenu pour entraîner des modeles ou le revendre.
# robots.txt n'est qu'une convention : les robots honnetes la respectent.
# Les plus agressifs sont bloques en dur (403) cote serveur, voir server.js.
# ----------------------------------------------------------------------
